#include <avr/io.h>
int main( void )
{
        DDRB |= ( 1 << PORTB5 );
        DDRB &= ~( 1 << PORTB0 );
        PORTB |= ( 1 << PORTB0 );
        while(42)
        {
                if( (PINB & ( 1 << PORTB0 )) )
                        PORTB |= ( 1 << PORTB5 );
                else
                        PORTB &= ~( 1 << PORTB5 );
        }
        return 0;
